    def attr(oc,at)
      o = find_objectclass(oc)
      case at.upcase
      when 'MUST'
        return o.must.collect { |e| e.to_s }
      when 'MAY'
        return o.may.collect { |e| e.to_s }
      when 'SUP'
        return o.sup.collect { |e| e.to_s }
      when 'NAME'
        return o.names.collect { |e| e.to_s }
      when 'DESC'
        return [o.desc]
      end
      return nil
    rescue LDAP::ResultError
      return nil
    end
