    def initialize(io, opt={})
      @io = io
      @opt = opt
      @mutex = Mutex.new
      @active_reqs = {}   # map message ID to thread object
      @binddn = nil
      @version = 3
      @logger = @opt[:logger] || $stderr
      @ssl = false

      startssl if @opt[:ssl_on_connect]
    end
