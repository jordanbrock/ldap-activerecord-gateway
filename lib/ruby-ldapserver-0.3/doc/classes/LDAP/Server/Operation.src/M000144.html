    def self.join_dn(elements)
      dn = ""
      elements.each do |elem|
        av = ""
        elem = [elem] if elem[0].is_a?(String)
        elem.each do |attr,val|
          av << "+" unless av == ""

          av << attr << "=" <<
                     val.sub(/^([# ])/, '\\\\\\1').
                     sub(/( )$/, '\\\\\\1').
                     gsub(/([,+"\\<>;])/, '\\\\\\1')
        end
        dn << "," unless dn == ""
        dn << av
      end
      dn
    end
