    def self.from_def(str, &blk)
      m = LDAPSyntaxDescription.match(str)
      raise LDAP::ResultError::InvalidAttributeSyntax,
        "Bad SyntaxTypeDescription #{str.inspect}" unless m
      new(m[1], m[2], :nhr=>(m[3] == 'TRUE'), :binary=>(m[4] == 'TRUE'), &blk)
    end
