    def initialize(oid, desc=nil, opt={}, &blk)
      @oid = oid
      @desc = desc
      @nhr = opt[:nhr]          # not human-readable?
      @binary = opt[:binary]    # binary encoding forced?
      @re = opt[:re]            # regular expression for parsing
      @def = nil
      instance_eval(&blk) if blk
    end
