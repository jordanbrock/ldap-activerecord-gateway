    def initialize(oid, names, syntax, desc=nil, obsolete=false, &blk)
      @oid = oid
      @names = names
      @names = [@names] unless @names.is_a?(Array)
      @desc = desc
      @obsolete = obsolete
      @syntax = LDAP::Server::Syntax.find(syntax)  # creates new obj if reqd
      @def = nil
      # initialization hook
      self.instance_eval(&blk) if blk
    end
