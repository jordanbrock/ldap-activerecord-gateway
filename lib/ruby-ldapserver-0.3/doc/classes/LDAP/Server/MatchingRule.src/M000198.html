    def to_def
      return @def if @def
      ans = "( #{@oid} "
      if names.nil? or @names.empty?
        # nothing
      elsif @names.size == 1
        ans << "NAME '#{@names[0]}' "
      else
        ans << "NAME ( "
        @names.each { |n| ans << "'#{n}' " }
        ans << ") "
      end
      ans << "DESC '#@desc' " if @desc
      ans << "OBSOLETE " if @obsolete
      ans << "SYNTAX #@syntax " if @syntax
      ans << ")"
      @def = ans
    end
